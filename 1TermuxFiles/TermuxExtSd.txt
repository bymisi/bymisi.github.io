xsd=$(cat "$file")

MicroSD="storage.txt"
if [ -s "$MicroSD" ]; then
   echo "Detected MicroSd:  $xsd"
   sleep 1;
else
   echo "MicroSd Card Not Detected"
   sleep 4 && exit
fi

DIR=~/$RD/storage/$xsd/termuxsd
if [ -d "$DIR" ]; then
   echo "termuxsd exists."
else
   echo "Creating Dir"
   mkdir ~/$RD/storage/$xsd/termuxsd
fi

LINK=~/$RD/$WD/$TD
if [ -h "$LINK" ]; then
   echo "LINK exists."
else
   echo "Creating Link"
   ln -s ~/$RD/storage/$xsd/termuxsd ~/$RD/$WD/$TD
fi

LINKSD=~/$RD/$WD/$xsd
if [ -h "$LINKSD" ]; then
   echo "LINKSD exists."
else
   echo "Creating Link"
   ln -s ~/$RD/storage/$xsd ~/$RD/$WD/extsd
fi

FILE=~/$TD/$TX
if [ -f "$FILE" ]; then
   echo "1.st FILE exists."
else
   echo "Creating 1.st File"
   echo "1.st" > ~/$TD/$TX
fi
rm storage.txt ;
